# Actually allow mu1=mu2
model
{
	# Log-uniform prior for the s.d.
	log_sigma ~ dunif(-5, 5)
	sigma <- exp(log_sigma)

	# First mean
	mu1 ~ dunif(-1000, 1000)

	# Prior on second mean, if it's different
	maybe ~ dunif(-1000, 1000)
	flag ~ dunif(0, 1)
	mu2 <- mu1 + (maybe - mu1)*step(flag - 0.5)

	# Sampling distribution/likelihood
	for(i in 1:N1)
	{
		y1[i] ~ dnorm(mu1, pow(sigma, -2))
	}
	for(i in 1:N2)
	{
		y2[i] ~ dnorm(mu2, pow(sigma, -2))
	}

}
